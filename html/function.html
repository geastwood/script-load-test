How function evaluated in Browser

Even if $ in function foo is not defined anywhere, it will not trigger a runtime error when the function is defined. Only when the foo is called, the global bar will be set a value. And error will be triggered.

        
        var bar; // defaut to undefined
        console.log(bar); // log undefined
        function foo() {

            bar = 'variable bar is only set when "foo" is called';
            $('some-selector').val('dummy value');
            alert('should not alert unless the function is called');

        }

        console.log(bar); // log undefined
        foo();
        console.log(bar); // log "variable bar is only set when "foo" is called"
